ORACLE - Verifica jobs por nome na base

set pages 999 lines 999;
COL OWNER FORM A7;
COL JOB_NAME FORM A30;
COL LAST_START_DATE FORM A31;
COL LAST_RUN_DURATION FORM A30;
COL REPEAT_INTERVAL FORM A24;
COL NEXT_RUN_DATE FORM A20;
select OWNER, JOB_NAME, LAST_START_DATE, LAST_RUN_DURATION, FAILURE_COUNT, NEXT_RUN_DATE, REPEAT_INTERVAL from DBA_SCHEDULER_JOBS 
where JOB_NAME='JOB_CARTAO_98001';
